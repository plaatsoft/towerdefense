35 void GRRLIB_SetBlend (const GRRLIB_blendMode blendmode) {
36  GRRLIB_Settings.blend = blendmode;
37  switch (GRRLIB_Settings.blend) {
38  case GRRLIB_BLEND_ALPHA:
39  GX_SetBlendMode(GX_BM_BLEND, GX_BL_SRCALPHA, GX_BL_INVSRCALPHA, GX_LO_CLEAR);
40  break;
41  case GRRLIB_BLEND_ADD:
42  GX_SetBlendMode(GX_BM_BLEND, GX_BL_SRCALPHA, GX_BL_DSTALPHA, GX_LO_CLEAR);
43  break;
44  case GRRLIB_BLEND_SCREEN:
45  GX_SetBlendMode(GX_BM_BLEND, GX_BL_SRCCLR, GX_BL_DSTALPHA, GX_LO_CLEAR);
46  break;
47  case GRRLIB_BLEND_MULTI:
48  GX_SetBlendMode(GX_BM_SUBSTRACT, GX_BL_SRCALPHA, GX_BL_INVSRCALPHA, GX_LO_CLEAR);
49  break;
50  case GRRLIB_BLEND_INV:
51  GX_SetBlendMode(GX_BM_BLEND, GX_BL_INVSRCCLR, GX_BL_INVSRCCLR, GX_LO_CLEAR);
52  break;
53  }
54 }
