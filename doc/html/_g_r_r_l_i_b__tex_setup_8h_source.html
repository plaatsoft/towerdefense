40 {
41  GRRLIB_texImg *my_texture = (struct GRRLIB_texImg *)calloc(1, sizeof(GRRLIB_texImg));
42 
43  if(my_texture != NULL) {
44  my_texture->data = memalign(32, h * w * 4);
45  my_texture->w = w;
46  my_texture->h = h;
47 
48  // Initialize the texture
49  memset(my_texture->data, '\0', (h * w) << 2);
50 
51  GRRLIB_SetHandle(my_texture, 0, 0);
52  GRRLIB_FlushTex(my_texture);
